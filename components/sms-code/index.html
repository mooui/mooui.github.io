SmsCode 验证码输入框 #

介绍 #

验证码输入

引入 #

// 入口引入
import { createApp } from 'vue';
import { MoSmsCode } from 'moxui';

const app = createApp();
app.use(MoSmsCode);


// 单文件组件引入
<!-- script:setup -->
import { MoSmsCode } from 'moxui';

<!-- template -->
<mo-sms-code
  v-model="code"
  title="输入验证码"
  tips="验证码已发送至 199 **** 0000"
  @finish="login"
>
</mo-sms-code>

代码演示 #

<mo-sms-code

import { ref } from "vue";
const code = ref("");

props #

参数说明类型默认值

length验证码长度number6
title标题string
tips提示string

Events #

事件名说明回调参数
update:modelValue验证码更新code
finish验证码输入完成code
